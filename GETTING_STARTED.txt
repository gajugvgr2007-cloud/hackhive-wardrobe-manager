━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

STEP 1: GET YOUR API KEYS (2 minutes)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. SUPABASE (Database):
   • Visit: https://supabase.com
   • Create a FREE account
   • Click "New Project"
   • Go to Settings > API
   • Copy your Project URL and Anon Key

2. OPENWEATHER (Weather):
   • Visit: https://openweathermap.org
   • Sign up for FREE account
   • Go to API keys page
   • Copy your API Key

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

STEP 2: CONFIGURE YOUR ENVIRONMENT (1 minute)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Create a file named .env.local in the project root and add:

    NEXT_PUBLIC_SUPABASE_URL=your_supabase_url
    NEXT_PUBLIC_SUPABASE_ANON_KEY=your_anon_key
    NEXT_PUBLIC_OPENWEATHER_API_KEY=your_weather_key

Replace the values with your actual keys from Step 1.

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

STEP 3: INSTALL DEPENDENCIES (1 minute)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

In your terminal, run:

    npm install

This installs all necessary packages including:
    • Next.js 13+
    • React 18
    • Tailwind CSS
    • Framer Motion
    • Supabase client
    • All UI components

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

STEP 4: RUN THE APPLICATION (instant)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

In your terminal, run:

    npm run dev

You should see:
    ▲ Next.js 13.5.1
    - Local:        http://localhost:3000

Open your browser and visit http://localhost:3000

You should see the beautiful Crypto Fashion dashboard!

Issue: "Can't find module"
Solution: Run `npm install` again

Issue: Weather shows "Location" instead of city
Solution: Allow browser permission for geolocation

Issue: See errors in console
Solution: Check .env.local has all three keys

Issue: Build fails
Solution: Delete .next folder and run `npm run build` again

Issue: API key errors
Solution: Verify your keys in .env.local are correct
